## Arquitetura da solução

```
/login   → email form → server action requestLoginOtp()
              ↓
/verify  → OTP input → signIn("credentials", { email, otp })
              ↓
         NextAuth authorize() → validateOtp() → retorna User
              ↓
         Sessão criada → redirect /dashboard ou /onboarding

/signup  → nome + email → server action requestSignupOtp()
              ↓
         Cria User (hasOnboarded=false) + OTP + email
              ↓
/verify  → mesma página → signIn → redirect /onboarding

Google   → signIn("google") → OAuth callback
              ↓
         PrismaAdapter cria User + Account automaticamente
              ↓
         signIn callback detecta novo usuário → redirect /onboarding
```

**OTP storage:** `VerificationToken` — `identifier=email`, `token=sha256(otp)`, `expires=now+10min`, `attempts=0..5`

### T-4: Serviço OTP — gerar, armazenar, validar
Tipo: feature
Agente: frontend

Criar `apps/web/src/lib/otp.ts`:

```typescript
import crypto from "crypto";
import { prisma } from "@fitflow/db";

const OTP_EXPIRY_MINUTES = 10;
const MAX_ATTEMPTS = 5;

function hashOtp(otp: string): string {
  return crypto.createHash("sha256").update(otp).digest("hex");
}

export function generateOtp(): string {
  return String(crypto.randomInt(100000, 999999));
}

export async function createOtp(email: string): Promise<string> {
  const otp = generateOtp();
  const expires = new Date(Date.now() + OTP_EXPIRY_MINUTES * 60 * 1000);

  await prisma.verificationToken.upsert({
    where: { identifier_token: { identifier: email, token: hashOtp(otp) } },
    update: { token: hashOtp(otp), expires, attempts: 0 },
    create: { identifier: email, token: hashOtp(otp), expires, attempts: 0 },
  });

  return otp;
}
```

> **Nota:** `upsert` acima tem limitação — o `where` de VerificationToken usa `identifier_token` compound unique. Como pode existir token antigo do mesmo email com hash diferente, usar `deleteMany` + `create` em vez de `upsert`:

```typescript
export async function createOtp(email: string): Promise<string> {
  const otp = generateOtp();
  const expires = new Date(Date.now() + OTP_EXPIRY_MINUTES * 60 * 1000);

  await prisma.verificationToken.deleteMany({ where: { identifier: email } });
  await prisma.verificationToken.create({
    data: { identifier: email, token: hashOtp(otp), expires, attempts: 0 },
  });

  return otp;
}

export async function validateOtp(
  email: string,
  otp: string
): Promise<{ valid: boolean; reason?: "invalid" | "expired" | "max_attempts" }> {
  const record = await prisma.verificationToken.findFirst({
    where: { identifier: email },
  });

  if (!record) return { valid: false, reason: "invalid" };
  if (record.expires < new Date()) {
    await prisma.verificationToken.deleteMany({ where: { identifier: email } });
    return { valid: false, reason: "expired" };
  }
  if (record.attempts >= MAX_ATTEMPTS) {
    await prisma.verificationToken.deleteMany({ where: { identifier: email } });
    return { valid: false, reason: "max_attempts" };
  }

  const isMatch = record.token === hashOtp(otp);

  if (!isMatch) {
    await prisma.verificationToken.updateMany({
      where: { identifier: email },
      data: { attempts: { increment: 1 } },
    });
    const remaining = MAX_ATTEMPTS - record.attempts - 1;
    if (remaining <= 0) {
      await prisma.verificationToken.deleteMany({ where: { identifier: email } });
    }
    return { valid: false, reason: "invalid" };
  }

  await prisma.verificationToken.deleteMany({ where: { identifier: email } });
  return { valid: true };
}
```

Critérios de Aceite:
- [ ] `generateOtp()` retorna string de 6 dígitos numéricos
- [ ] `createOtp(email)` deleta tokens anteriores do mesmo email antes de criar novo
- [ ] `validateOtp` retorna `valid: false, reason: "expired"` para tokens expirados
- [ ] `validateOtp` retorna `valid: false, reason: "max_attempts"` após 5 tentativas erradas
- [ ] `validateOtp` deleta o token após uso bem-sucedido
- [ ] Sem dependência de frameworks além de `crypto` (nativo Node) e `@fitflow/db`

### T-5: NextAuth — configurar providers Google e Credentials
Tipo: feature
Agente: frontend

Substituir o conteúdo de `apps/web/src/lib/auth.ts`:

```typescript
import NextAuth from "next-auth";
import Google from "next-auth/providers/google";
import Credentials from "next-auth/providers/credentials";
import { PrismaAdapter } from "@auth/prisma-adapter";
import { prisma } from "@fitflow/db";
import { validateOtp } from "@/lib/otp";

const nextAuth = NextAuth({
  adapter: PrismaAdapter(prisma),
  providers: [
    Google({
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    }),
    Credentials({
      credentials: {
        email: { type: "email" },
        otp: { type: "text" },
      },
      async authorize(credentials) {
        const email = credentials?.email as string;
        const otp = credentials?.otp as string;
        if (!email || !otp) return null;

        const result = await validateOtp(email, otp);
        if (!result.valid) return null;

        const user = await prisma.user.findUnique({ where: { email } });
        if (!user) return null;

        // Marcar email como verificado no primeiro login OTP
        if (!user.emailVerified) {
          await prisma.user.update({
            where: { id: user.id },
            data: { emailVerified: new Date() },
          });
        }

        return { id: user.id, email: user.email, name: user.name, hasOnboarded: user.hasOnboarded };
      },
    }),
  ],
  session: { strategy: "database" },
  pages: {
    signIn: "/login",
  },
  callbacks: {
    async session({ session, user }) {
      if (user) {
        const dbUser = await prisma.user.findUnique({
          where: { id: user.id },
          select: { hasOnboarded: true },
        });
        (session.user as any).hasOnboarded = dbUser?.hasOnboarded ?? false;
        (session.user as any).id = user.id;
      }
      return session;
    },
    async signIn({ user, account, profile }) {
      // Google: auto-criar usuário se necessário (PrismaAdapter já cria)
      // Garantir que name não seja nulo (Google sempre fornece)
      if (account?.provider === "google" && profile?.name && user.id) {
        await prisma.user.update({
          where: { id: user.id },
          data: { name: profile.name },
        }).catch(() => {}); // silencia se usuário não existir ainda
      }
      return true;
    },
  },
});

export const { handlers } = nextAuth;

// eslint-disable-next-line @typescript-eslint/no-explicit-any
export const auth: (...args: any[]) => any = nextAuth.auth;
// eslint-disable-next-line @typescript-eslint/no-explicit-any
export const signIn: (...args: any[]) => any = nextAuth.signIn;
// eslint-disable-next-line @typescript-eslint/no-explicit-any
export const signOut: (...args: any[]) => any = nextAuth.signOut;
```

### T-6: Server Actions — requestLoginOtp e requestSignupOtp
Tipo: feature
Agente: frontend

Criar `apps/web/src/app/(auth)/actions.ts`:

```typescript
"use server";

import { prisma } from "@fitflow/db";
import { createOtp } from "@/lib/otp";
import { sendOtpEmail } from "@/lib/email";

export async function requestLoginOtp(email: string): Promise<{ success: boolean; error?: string }> {
  const user = await prisma.user.findUnique({ where: { email } });
  if (!user) return { success: false, error: "Email não encontrado. Crie uma conta primeiro." };

  const otp = await createOtp(email);
  await sendOtpEmail(email, user.name, otp);

  return { success: true };
}

export async function requestSignupOtp(
  name: string,
  email: string
): Promise<{ success: boolean; error?: string }> {
  const existing = await prisma.user.findUnique({ where: { email } });
  if (existing) return { success: false, error: "Email já cadastrado. Faça login." };

  await prisma.user.create({
    data: { name, email, hasOnboarded: false },
  });

  const otp = await createOtp(email);
  await sendOtpEmail(email, name, otp);

  return { success: true };
}

export async function resendOtp(email: string): Promise<{ success: boolean; error?: string }> {
  const user = await prisma.user.findUnique({ where: { email } });
  if (!user) return { success: false, error: "Email não encontrado." };

  const otp = await createOtp(email);
  await sendOtpEmail(email, user.name, otp);

  return { success: true };
}
```

Critérios de Aceite:
- [ ] `requestSignupOtp` rejeita email já existente com mensagem clara
- [ ] `requestLoginOtp` rejeita email não cadastrado com mensagem clara
- [ ] Ambas as actions criam OTP e enviam email quando válidas
- [ ] `resendOtp` reutiliza `createOtp` (que já deleta token anterior)
- [ ] Arquivo usa `"use server"` no topo

### T-7: Middleware — proteção de rotas
Tipo: feature
Agente: frontend

Criar `apps/web/src/middleware.ts`:

```typescript
import { auth } from "@/lib/auth";
import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";

const PUBLIC_PATHS = ["/", "/login", "/signup", "/verify"];

export default async function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  const isPublic =
    PUBLIC_PATHS.includes(pathname) ||
    pathname.startsWith("/api/auth") ||
    pathname.startsWith("/_next") ||
    pathname.startsWith("/favicon");

  if (isPublic) return NextResponse.next();

  const session = await auth();

  if (!session?.user) {
    return NextResponse.redirect(new URL("/login", req.url));
  }

  return NextResponse.next();
}

export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],
};
```

Critérios de Aceite:
- [ ] Acessar `/dashboard` sem sessão redireciona para `/login`
- [ ] Acessar `/login` sem sessão retorna 200
- [ ] Acessar `/api/auth/signin` não é bloqueado pelo middleware
- [ ] Sessão válida permite acesso a `/dashboard`
